021package com.samaxes.maven.plugin.minify;
022
023import java.util.ArrayList;
024import java.util.Collection;
025import java.util.concurrent.Callable;
026import java.util.concurrent.ExecutorService;
027import java.util.concurrent.Executors;
028import java.util.concurrent.TimeUnit;
029
030import org.apache.maven.plugin.AbstractMojo;
031import org.apache.maven.plugin.MojoExecutionException;
032import org.apache.maven.plugin.MojoFailureException;
033
034/**
035 * Goal which combines and minifies CSS and JavaScript files.
036 *
037 * @goal minify
038 * @phase process-resources
039 */
040public class MinifyMojo extends AbstractMojo {
041
042    /**
043     * Webapp source directory.
044     *
045     * @parameter expression="${webappSourceDir}" default-value="${basedir}/src/main/webapp"
046     */
047    private String webappSourceDir;
048
049    /**
050     * Webapp target directory.
051     *
052     * @parameter expression="${webappTargetDir}" default-value="${project.build.directory}/${project.build.finalName}"
053     */
054    private String webappTargetDir;
055
056    /**
057     * CSS source directory.
058     *
059     * @parameter expression="${cssSourceDir}" default-value="css"
060     */
061    private String cssSourceDir;
062
063    /**
064     * JavaScript source directory.
065     *
066     * @parameter expression="${jsSourceDir}" default-value="js"
067     */
068    private String jsSourceDir;
069
070    /**
071     * CSS source filenames list.
072     *
073     * @parameter expression="${cssSourceFiles}" alias="cssFiles"
074     */
075    private ArrayList<String> cssSourceFiles;
076
077    /**
078     * JavaScript source filenames list.
079     *
080     * @parameter expression="${jsSourceFiles}" alias="jsFiles"
081     */
082    private ArrayList<String> jsSourceFiles;
083
084    /**
085     * CSS files to include. Specified as fileset patterns which are relative to the CSS source directory.
086     *
087     * @parameter expression="${cssSourceIncludes}" alias="cssIncludes"
088     * @since 1.2
089     */
090    private ArrayList<String> cssSourceIncludes;
091
092    /**
093     * JavaScript files to include. Specified as fileset patterns which are relative to the JavaScript source directory.
094     *
095     * @parameter expression="${jsSourceIncludes}" alias="jsIncludes"
096     * @since 1.2
097     */
098    private ArrayList<String> jsSourceIncludes;
099
100    /**
101     * CSS files to exclude. Specified as fileset patterns which are relative to the CSS source directory.
102     *
103     * @parameter expression="${cssSourceExcludes}" alias="cssExcludes"
104     * @since 1.2
105     */
106    private ArrayList<String> cssSourceExcludes;
107
108    /**
109     * JavaScript files to exclude. Specified as fileset patterns which are relative to the JavaScript source directory.
110     *
111     * @parameter expression="${jsSourceExcludes}" alias="jsExcludes"
112     * @since 1.2
113     */
114    private ArrayList<String> jsSourceExcludes;
115
116    /**
117     * CSS target directory.
118     *
119     * @parameter expression="${cssTargetDir}" default-value="css"
120     * @since 1.3.2
121     */
122    private String cssTargetDir;
123
124    /**
125     * JavaScript target directory.
126     *
127     * @parameter expression="${jsTargetDir}" default-value="js"
128     * @since 1.3.2
129     */
130    private String jsTargetDir;
131
132    /**
133     * CSS output filename.
134     *
135     * @parameter expression="${cssFinalFile}" default-value="style.css"
136     */
137    private String cssFinalFile;
138
139    /**
140     * JavaScript output filename.
141     *
142     * @parameter expression="${jsFinalFile}" default-value="script.js"
143     */
144    private String jsFinalFile;
145
146    /**
147     * The output filename suffix.
148     *
149     * @parameter expression="${suffix}" default-value=".min"
150     * @since 1.3.2
151     */
152    private String suffix;
153
154    /**
155     * <p>
156     * If a supported character set is specified, it will be used to read the input file. Otherwise, it will assume that
157     * the platform's default character set is being used. The output file is encoded using the same character set.
158     * </p>
159     * <p>
160     * See the <a href="http://www.iana.org/assignments/character-sets">IANA Charset Registry</a> for a list of valid
161     * encoding types.
162     * </p>
163     *
164     * @parameter expression="${charset}"
165     * @since 1.3.2
166     */
167    private String charset;
168
169    /**
170     * Some source control tools don't like files containing lines longer than, say 8000 characters. The linebreak
171     * option is used in that case to split long lines after a specific column. It can also be used to make the code
172     * more readable, easier to debug (especially with the MS Script Debugger). Specify 0 to get a line break after each
173     * semi-colon in JavaScript, and after each rule in CSS. Specify -1 to disallow line breaks.
174     *
175     * @parameter expression="${linebreak}" default-value="-1"
176     */
177    private int linebreak;
178
179    /**
180     * JAVASCRIPT ONLY OPTION!<br/>
181     * Minify only. Do not obfuscate local symbols.
182     *
183     * @parameter expression="${munge}" default-value="false"
184     */
185    private boolean nomunge;
186
187    /**
188     * JAVASCRIPT ONLY OPTION!<br/>
189     * Display informational messages and warnings.
190     *
191     * @parameter expression="${verbose}" default-value="false"
192     */
193    private boolean verbose;
194
195    /**
196     * JAVASCRIPT ONLY OPTION!<br/>
197     * Preserve unnecessary semicolons (such as right before a '}'). This option is useful when compressed code has to
198     * be run through JSLint (which is the case of YUI for example).
199     *
200     * @parameter expression="${preserveAllSemiColons}" default-value="false"
201     */
202    private boolean preserveAllSemiColons;
203
204    /**
205     * JAVASCRIPT ONLY OPTION!<br/>
206     * Disable all the built-in micro optimizations.
207     *
208     * @parameter expression="${disableOptimizations}" default-value="false"
209     */
210    private boolean disableOptimizations;
211
212    /**
213     * Size of the buffer used to read source files.
214     *
215     * @parameter expression="${bufferSize}" default-value="4096"
216     */
217    private int bufferSize;
218
219    /**
220     * Maximum execution time in seconds.
221     *
222     * @parameter expression="${timeout}" default-value="30"
223     * @since 1.5
224     */
225    private long timeout;
226
227    /**
228     * Executed when the goal is invoked, it will first invoke a parallel lifecycle, ending at the given phase.
229     */
230    public void execute() throws MojoExecutionException, MojoFailureException {
231        Collection<Callable<Object>> processFilesTasks = new ArrayList<Callable<Object>>();
232        processFilesTasks.add(new ProcessCSSFilesTask(getLog(), bufferSize, webappSourceDir, webappTargetDir,
233                cssSourceDir, cssSourceFiles, cssSourceIncludes, cssSourceExcludes, cssTargetDir, cssFinalFile, suffix,
234                charset, linebreak));
235        processFilesTasks.add(new ProcessJSFilesTask(getLog(), bufferSize, webappSourceDir, webappTargetDir,
236                jsSourceDir, jsSourceFiles, jsSourceIncludes, jsSourceExcludes, jsTargetDir, jsFinalFile, suffix,
237                charset, linebreak, !nomunge, verbose, preserveAllSemiColons, disableOptimizations));
238
239        ExecutorService executor = Executors.newFixedThreadPool(2);
240        try {
241            executor.invokeAll(processFilesTasks);
242            executor.shutdown();
243            executor.awaitTermination(timeout, TimeUnit.SECONDS);
244        } catch (InterruptedException e) {
245            getLog().error(e.getMessage(), e);
246        }
247    }
248}
