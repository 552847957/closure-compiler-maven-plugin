021package com.samaxes.maven.plugin.minify;
022
023import java.io.FileReader;
024import java.io.FileWriter;
025import java.io.IOException;
026import java.io.Reader;
027import java.io.Writer;
028import java.util.List;
029
030import org.apache.maven.plugin.logging.Log;
031
032import com.samaxes.maven.plugin.common.JavaScriptErrorReporter;
033import com.yahoo.platform.yui.compressor.JavaScriptCompressor;
034
035/**
036 * Task for merging and compressing JavaScript files.
037 */
038public class ProcessJSFilesTask extends ProcessFilesTask {
039
040    private boolean munge;
041
042    private boolean verbose;
043
044    private boolean preserveAllSemiColons;
045
046    private boolean disableOptimizations;
047
048    /**
049     * Task constructor.
050     * 
051     * @param log Maven plugin log
052     * @param bufferSize size of the buffer used to read source files.
053     * @param webappSourceDir web resources source directory
054     * @param webappTargetDir web resources target directory
055     * @param inputDir directory containing source files
056     * @param sourceFiles list of source files to include
057     * @param sourceIncludes list of source files to include
058     * @param sourceExcludes list of source files to exclude
059     * @param outputDir directory to write the final file
060     * @param finalFilename final filename
061     * @param suffix final filename suffix
062     * @param charset if a character set is specified, a byte-to-char variant allows the encoding to be selected.
063     *        Otherwise, only byte-to-byte operations are used
064     * @param linebreak split long lines after a specific column
065     * @param munge minify only
066     * @param verbose display informational messages and warnings
067     * @param preserveAllSemiColons preserve unnecessary semicolons
068     * @param disableOptimizations disable all the built-in micro optimizations
069     */
070    public ProcessJSFilesTask(Log log, Integer bufferSize, String webappSourceDir, String webappTargetDir,
071            String inputDir, List<String> sourceFiles, List<String> sourceIncludes, List<String> sourceExcludes,
072            String outputDir, String finalFilename, String suffix, String charset, int linebreak, boolean munge,
073            boolean verbose, boolean preserveAllSemiColons, boolean disableOptimizations) {
074        super(log, bufferSize, webappSourceDir, webappTargetDir, inputDir, sourceFiles, sourceIncludes, sourceExcludes,
075                outputDir, finalFilename, suffix, charset, linebreak);
076
077        this.munge = munge;
078        this.verbose = verbose;
079        this.preserveAllSemiColons = preserveAllSemiColons;
080        this.disableOptimizations = disableOptimizations;
081    }
082
083    /**
084     * Minifies JavaScript file.
085     */
086    protected void minify() {
087        if (minifiedFile != null) {
088            try {
089                log.info("Creating minified file [" + minifiedFile.getName() + "]");
090                Reader reader = new FileReader(mergedFile);
091                Writer writer = new FileWriter(minifiedFile);
092                JavaScriptCompressor compressor = new JavaScriptCompressor(reader, new JavaScriptErrorReporter(log,
093                        mergedFile.getName()));
094
095                compressor.compress(writer, linebreak, munge, verbose, preserveAllSemiColons, disableOptimizations);
096                reader.close();
097                writer.close();
098            } catch (IOException e) {
099                log.error(e.getMessage(), e);
100            }
101        }
102    }
103}
