021package com.samaxes.maven.plugin.minify;
022
023import java.io.FileReader;
024import java.io.FileWriter;
025import java.io.IOException;
026import java.io.Reader;
027import java.io.Writer;
028import java.util.List;
029
030import org.apache.maven.plugin.logging.Log;
031
032import com.yahoo.platform.yui.compressor.CssCompressor;
033
034/**
035 * Task for merging and compressing CSS files.
036 */
037public class ProcessCSSFilesTask extends ProcessFilesTask {
038
039    /**
040     * Task constructor.
041     * 
042     * @param log Maven plugin log
043     * @param bufferSize size of the buffer used to read source files.
044     * @param webappSourceDir web resources source directory
045     * @param webappTargetDir web resources target directory
046     * @param inputDir directory containing source files
047     * @param sourceFiles list of source files to include
048     * @param sourceIncludes list of source files to include
049     * @param sourceExcludes list of source files to exclude
050     * @param outputDir directory to write the final file
051     * @param finalFilename final filename
052     * @param suffix final filename suffix
053     * @param charset if a character set is specified, a byte-to-char variant allows the encoding to be selected.
054     *        Otherwise, only byte-to-byte operations are used
055     * @param linebreak split long lines after a specific column
056     */
057    public ProcessCSSFilesTask(Log log, Integer bufferSize, String webappSourceDir, String webappTargetDir,
058            String inputDir, List<String> sourceFiles, List<String> sourceIncludes, List<String> sourceExcludes,
059            String outputDir, String finalFilename, String suffix, String charset, int linebreak) {
060        super(log, bufferSize, webappSourceDir, webappTargetDir, inputDir, sourceFiles, sourceIncludes, sourceExcludes,
061                outputDir, finalFilename, suffix, charset, linebreak);
062    }
063
064    /**
065     * Minifies CSS file.
066     */
067    protected void minify() {
068        if (minifiedFile != null) {
069            try {
070                log.info("Creating minified file [" + minifiedFile.getName() + "]");
071                Reader reader = new FileReader(mergedFile);
072                Writer writer = new FileWriter(minifiedFile);
073                CssCompressor compressor = new CssCompressor(reader);
074
075                compressor.compress(writer, linebreak);
076                reader.close();
077                writer.close();
078            } catch (IOException e) {
079                log.error(e.getMessage(), e);
080            }
081        }
082    }
083}
