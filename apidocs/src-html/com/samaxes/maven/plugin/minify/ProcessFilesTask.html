021package com.samaxes.maven.plugin.minify;
022
023import java.io.File;
024import java.io.FileOutputStream;
025import java.io.IOException;
026import java.io.InputStream;
027import java.io.InputStreamReader;
028import java.io.OutputStream;
029import java.io.OutputStreamWriter;
030import java.io.SequenceInputStream;
031import java.util.ArrayList;
032import java.util.Collections;
033import java.util.List;
034
035import org.apache.maven.plugin.logging.Log;
036import org.codehaus.plexus.util.DirectoryScanner;
037import org.codehaus.plexus.util.FileUtils;
038import org.codehaus.plexus.util.IOUtil;
039
040import com.samaxes.maven.plugin.common.FilenameComparator;
041import com.samaxes.maven.plugin.common.ListOfFiles;
042
043/**
044 * Abstract class for merging and compressing a files list.
045 */
046public abstract class ProcessFilesTask implements Runnable {
047
048    protected Log log;
049
050    protected int linebreak;
051
052    protected File mergedFile;
053
054    protected File minifiedFile;
055
056    private Integer bufferSize;
057
058    private String charset;
059
060    private List<File> files = new ArrayList<File>();
061
062    /**
063     * Task constructor.
064     * 
065     * @param log Maven plugin log
066     * @param bufferSize size of the buffer used to read source files.
067     * @param webappSourceDir web resources source directory
068     * @param webappTargetDir web resources target directory
069     * @param inputDir directory containing source files
070     * @param sourceFiles list of source files to include
071     * @param sourceIncludes list of source files to include
072     * @param sourceExcludes list of source files to exclude
073     * @param outputDir directory to write the final file
074     * @param finalFilename final filename
075     * @param suffix final filename suffix
076     * @param charset if a character set is specified, a byte-to-char variant allows the encoding to be selected.
077     *        Otherwise, only byte-to-byte operations are used
078     * @param linebreak split long lines after a specific column
079     */
080    public ProcessFilesTask(Log log, Integer bufferSize, String webappSourceDir, String webappTargetDir,
081            String inputDir, List<String> sourceFiles, List<String> sourceIncludes, List<String> sourceExcludes,
082            String outputDir, String finalFilename, String suffix, String charset, int linebreak) {
083        this.log = log;
084        this.linebreak = linebreak;
085        this.bufferSize = bufferSize;
086        this.charset = charset;
087
088        File sourceDir = new File(webappSourceDir.concat(File.separator).concat(inputDir));
089        File targetDir = new File(webappTargetDir.concat(File.separator).concat(outputDir));
090
091        for (String sourceFile : sourceFiles) {
092            logNewSourceFile(finalFilename, sourceFile);
093            files.add(new File(sourceDir, sourceFile));
094        }
095
096        for (File sourceInclude : getFilesToInclude(sourceDir, sourceIncludes, sourceExcludes)) {
097            if (!files.contains(sourceInclude)) {
098                logNewSourceFile(finalFilename, sourceInclude.getName());
099                files.add(sourceInclude);
100            }
101        }
102
103        if (!files.isEmpty() && (targetDir.exists() || targetDir.mkdirs())) {
104            this.mergedFile = new File(targetDir, finalFilename);
105
106            String extension = ".".concat(FileUtils.getExtension(this.mergedFile.getName()));
107            this.minifiedFile = new File(targetDir, this.mergedFile.getName().replace(extension,
108                    suffix.concat(extension)));
109        } else if (!sourceFiles.isEmpty() || !sourceIncludes.isEmpty()) {
110            // The 'files' list will be empty if the source file paths or names added to the project's POM are wrong.
111            log.error("An error has occurred while loading source files."
112                    + " Please check your source directory path and source file names.");
113        }
114    }
115
116    /**
117     * Method executed by the thread.
118     */
119    public void run() {
120        mergeFiles();
121        minify();
122    }
123
124    /**
125     * Logs an addition of a new source file.
126     * 
127     * @param finalFilename the final file name
128     * @param sourceFilename the source file name
129     */
130    private void logNewSourceFile(String finalFilename, String sourceFilename) {
131        if (finalFilename.equalsIgnoreCase(sourceFilename)) {
132            log.info("Please be cautious when using the source file name for the final file.");
133            // throw new IllegalArgumentException("Source file should not have the same name as final file ["
134            // + sourceFilename + "]");
135        }
136        log.debug("Adding source file [" + sourceFilename + "]");
137    }
138
139    /**
140     * Returns the files to copy. Default exclusions are used when the excludes list is empty.
141     * 
142     * @param baseDir the base directory to start from
143     * @param includes list of source files to include
144     * @param excludes list of source files to exclude
145     * @return the files to copy
146     */
147    private List<File> getFilesToInclude(File baseDir, List<String> includes, List<String> excludes) {
148        List<File> includedFiles = new ArrayList<File>();
149
150        if (includes != null && !includes.isEmpty()) {
151            DirectoryScanner scanner = new DirectoryScanner();
152
153            scanner.setIncludes(includes.toArray(new String[] {}));
154            scanner.setExcludes(excludes.toArray(new String[] {}));
155            scanner.addDefaultExcludes();
156            scanner.setBasedir(baseDir);
157            scanner.scan();
158
159            for (String includedFilename : scanner.getIncludedFiles()) {
160                includedFiles.add(new File(baseDir, includedFilename));
161            }
162
163            Collections.sort(includedFiles, new FilenameComparator());
164        }
165
166        return includedFiles;
167    }
168
169    /**
170     * Merges files list.
171     */
172    private void mergeFiles() {
173        if (mergedFile != null) {
174            ListOfFiles listOfFiles = new ListOfFiles(log, files);
175
176            try {
177                log.info("Creating merged file [" + mergedFile.getName() + "]");
178                InputStream sequence = new SequenceInputStream(listOfFiles);
179                OutputStream out = new FileOutputStream(mergedFile);
180
181                if (charset == null) {
182                    IOUtil.copy(sequence, out, bufferSize);
183                } else {
184                    InputStreamReader sequenceReader = new InputStreamReader(sequence, charset);
185                    OutputStreamWriter outWriter = new OutputStreamWriter(out, charset);
186
187                    IOUtil.copy(sequenceReader, outWriter, bufferSize);
188                    IOUtil.close(sequenceReader);
189                    IOUtil.close(outWriter);
190                }
191
192                IOUtil.close(sequence);
193                IOUtil.close(out);
194            } catch (IOException e) {
195                log.error("An error has occurred while concatenating files.", e);
196            }
197        }
198    }
199
200    /**
201     * Minifies source file.
202     */
203    abstract void minify();
204}
