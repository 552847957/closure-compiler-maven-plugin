021package com.samaxes.maven.minify.plugin;
022
023import java.util.ArrayList;
024import java.util.Collection;
025import java.util.concurrent.ExecutorService;
026import java.util.concurrent.Executors;
027import java.util.concurrent.TimeUnit;
028
029import org.apache.maven.plugin.AbstractMojo;
030import org.apache.maven.plugin.MojoExecutionException;
031import org.apache.maven.plugin.MojoFailureException;
032
033/**
034 * Goal which combines and minifies CSS and JavaScript files.
035 *
036 * @goal minify
037 * @phase process-resources
038 */
039public class MinifyMojo extends AbstractMojo {
040
041    /**
042     * Webapp source directory.
043     *
044     * @parameter expression="${webappSourceDir}" default-value="${basedir}/src/main/webapp"
045     */
046    private String webappSourceDir;
047
048    /**
049     * Webapp target directory.
050     *
051     * @parameter expression="${webappTargetDir}" default-value="${project.build.directory}/${project.build.finalName}"
052     */
053    private String webappTargetDir;
054
055    /**
056     * CSS source directory.
057     *
058     * @parameter expression="${cssSourceDir}" default-value="css"
059     */
060    private String cssSourceDir;
061
062    /**
063     * JavaScript source directory.
064     *
065     * @parameter expression="${jsSourceDir}" default-value="js"
066     */
067    private String jsSourceDir;
068
069    /**
070     * CSS source filenames list.
071     *
072     * @parameter expression="${cssSourceFiles}" alias="cssFiles"
073     */
074    private ArrayList<String> cssSourceFiles;
075
076    /**
077     * JavaScript source filenames list.
078     *
079     * @parameter expression="${jsSourceFiles}" alias="jsFiles"
080     */
081    private ArrayList<String> jsSourceFiles;
082
083    /**
084     * CSS files to include. Specified as fileset patterns which are relative to the CSS source directory.
085     *
086     * @parameter expression="${cssSourceIncludes}" alias="cssIncludes"
087     * @since 1.2
088     */
089    private ArrayList<String> cssSourceIncludes;
090
091    /**
092     * JavaScript files to include. Specified as fileset patterns which are relative to the JavaScript source directory.
093     *
094     * @parameter expression="${jsSourceIncludes}" alias="jsIncludes"
095     * @since 1.2
096     */
097    private ArrayList<String> jsSourceIncludes;
098
099    /**
100     * CSS files to exclude. Specified as fileset patterns which are relative to the CSS source directory.
101     *
102     * @parameter expression="${cssSourceExcludes}" alias="cssExcludes"
103     * @since 1.2
104     */
105    private ArrayList<String> cssSourceExcludes;
106
107    /**
108     * JavaScript files to exclude. Specified as fileset patterns which are relative to the JavaScript source directory.
109     *
110     * @parameter expression="${jsSourceExcludes}" alias="jsExcludes"
111     * @since 1.2
112     */
113    private ArrayList<String> jsSourceExcludes;
114
115    /**
116     * CSS target directory.
117     *
118     * @parameter expression="${cssTargetDir}" default-value="css"
119     * @since 1.3.2
120     */
121    private String cssTargetDir;
122
123    /**
124     * JavaScript target directory.
125     *
126     * @parameter expression="${jsTargetDir}" default-value="js"
127     * @since 1.3.2
128     */
129    private String jsTargetDir;
130
131    /**
132     * CSS output filename.
133     *
134     * @parameter expression="${cssFinalFile}" default-value="style.css"
135     */
136    private String cssFinalFile;
137
138    /**
139     * JavaScript output filename.
140     *
141     * @parameter expression="${jsFinalFile}" default-value="script.js"
142     */
143    private String jsFinalFile;
144
145    /**
146     * The output filename suffix.
147     *
148     * @parameter expression="${suffix}" default-value=".min"
149     * @since 1.3.2
150     */
151    private String suffix;
152
153    /**
154     * <p>
155     * If a supported character set is specified, it will be used to read the input file. Otherwise, it will assume that
156     * the platform's default character set is being used. The output file is encoded using the same character set.
157     * </p>
158     * <p>
159     * See the <a href="http://www.iana.org/assignments/character-sets">IANA Charset Registry</a> for a list of valid
160     * encoding types.
161     * </p>
162     *
163     * @parameter expression="${charset}"
164     * @since 1.3.2
165     */
166    private String charset;
167
168    /**
169     * Some source control tools don't like files containing lines longer than, say 8000 characters. The linebreak
170     * option is used in that case to split long lines after a specific column. It can also be used to make the code
171     * more readable, easier to debug (especially with the MS Script Debugger). Specify 0 to get a line break after each
172     * semi-colon in JavaScript, and after each rule in CSS. Specify -1 to disallow line breaks.
173     *
174     * @parameter expression="${linebreak}" default-value="-1"
175     */
176    private int linebreak;
177
178    /**
179     * JAVASCRIPT ONLY OPTION!<br/>
180     * Minify only. Do not obfuscate local symbols.
181     *
182     * @parameter expression="${munge}" default-value="false"
183     */
184    private boolean nomunge;
185
186    /**
187     * JAVASCRIPT ONLY OPTION!<br/>
188     * Display informational messages and warnings.
189     *
190     * @parameter expression="${verbose}" default-value="false"
191     */
192    private boolean verbose;
193
194    /**
195     * JAVASCRIPT ONLY OPTION!<br/>
196     * Preserve unnecessary semicolons (such as right before a '}'). This option is useful when compressed code has to
197     * be run through JSLint (which is the case of YUI for example).
198     *
199     * @parameter expression="${preserveAllSemiColons}" default-value="false"
200     */
201    private boolean preserveAllSemiColons;
202
203    /**
204     * JAVASCRIPT ONLY OPTION!<br/>
205     * Disable all the built-in micro optimizations.
206     *
207     * @parameter expression="${disableOptimizations}" default-value="false"
208     */
209    private boolean disableOptimizations;
210
211    /**
212     * Size of the buffer used to read source files.
213     *
214     * @parameter expression="${bufferSize}" default-value="4096"
215     */
216    private int bufferSize;
217
218    /**
219     * Maximum execution time in seconds.
220     *
221     * @parameter expression="${timeout}" default-value="30"
222     * @since 1.5
223     */
224    private long timeout;
225
226    /**
227     * Show source file paths in log output.
228     *
229     * @parameter expression="${debug}" default-value="false"
230     * @since 1.5.2
231     */
232    private boolean debug;
233
234    /**
235     * Skip the merge step. Minification will be applied to each source file individually.
236     *
237     * @parameter expression="${skipMerge}" default-value="false"
238     * @since 1.5.2
239     */
240    private boolean skipMerge;
241
242    /**
243     * Skip the minify step. Useful when merging files that are already minified.
244     *
245     * @parameter expression="${skipMinify}" default-value="false"
246     * @since 1.5.2
247     */
248    private boolean skipMinify;
249
250    /**
251     * Executed when the goal is invoked, it will first invoke a parallel lifecycle, ending at the given phase.
252     */
253    public void execute() throws MojoExecutionException, MojoFailureException {
254        if (skipMerge && skipMinify) {
255            getLog().warn("Both merge and minify steps are configured to be skipped.");
256            return;
257        }
258
259        Collection<ProcessFilesTask> processFilesTasks = new ArrayList<ProcessFilesTask>();
260        processFilesTasks.add(new ProcessCSSFilesTask(getLog(), bufferSize, debug, skipMerge, skipMinify,
261                webappSourceDir, webappTargetDir, cssSourceDir, cssSourceFiles, cssSourceIncludes, cssSourceExcludes,
262                cssTargetDir, cssFinalFile, suffix, charset, linebreak));
263        processFilesTasks.add(new ProcessJSFilesTask(getLog(), bufferSize, debug, skipMerge, skipMinify,
264                webappSourceDir, webappTargetDir, jsSourceDir, jsSourceFiles, jsSourceIncludes, jsSourceExcludes,
265                jsTargetDir, jsFinalFile, suffix, charset, linebreak, !nomunge, verbose, preserveAllSemiColons,
266                disableOptimizations));
267
268        ExecutorService executor = Executors.newFixedThreadPool(2);
269        try {
270            executor.invokeAll(processFilesTasks);
271            executor.shutdown();
272            executor.awaitTermination(timeout, TimeUnit.SECONDS);
273        } catch (InterruptedException e) {
274            getLog().error(e.getMessage(), e);
275        }
276    }
277}
