019package com.samaxes.maven.minify.plugin;
020
021import java.io.File;
022import java.io.FileInputStream;
023import java.io.FileOutputStream;
024import java.io.IOException;
025import java.io.InputStream;
026import java.io.InputStreamReader;
027import java.io.OutputStream;
028import java.io.OutputStreamWriter;
029import java.util.List;
030
031import org.apache.maven.plugin.logging.Log;
032
033import com.google.common.collect.Lists;
034import com.google.javascript.jscomp.Compiler;
035import com.google.javascript.jscomp.CompilerOptions;
036import com.google.javascript.jscomp.SourceFile;
037import com.google.javascript.rhino.head.EvaluatorException;
038import com.samaxes.maven.minify.common.ClosureConfig;
039import com.samaxes.maven.minify.common.JavaScriptErrorReporter;
040import com.samaxes.maven.minify.common.YuiConfig;
041import com.samaxes.maven.minify.plugin.MinifyMojo.Engine;
042import com.yahoo.platform.yui.compressor.JavaScriptCompressor;
043
044/**
045 * Task for merging and compressing JavaScript files.
046 */
047public class ProcessJSFilesTask extends ProcessFilesTask {
048
049    private final ClosureConfig closureConfig;
050
051    /**
052     * Task constructor.
053     *
054     * @param log Maven plugin log
055     * @param verbose display additional info
056     * @param bufferSize size of the buffer used to read source files
057     * @param charset if a character set is specified, a byte-to-char variant allows the encoding to be selected.
058     *        Otherwise, only byte-to-byte operations are used
059     * @param suffix final file name suffix
060     * @param nosuffix whether to use a suffix for the minified file name or not
061     * @param skipMerge whether to skip the merge step or not
062     * @param skipMinify whether to skip the minify step or not
063     * @param webappSourceDir web resources source directory
064     * @param webappTargetDir web resources target directory
065     * @param inputDir directory containing source files
066     * @param sourceFiles list of source files to include
067     * @param sourceIncludes list of source files to include
068     * @param sourceExcludes list of source files to exclude
069     * @param outputDir directory to write the final file
070     * @param outputFilename the output file name
071     * @param engine minify processor engine selected
072     * @param yuiConfig YUI Compressor configuration
073     * @param closureConfig Google Closure Compiler configuration
074     */
075    public ProcessJSFilesTask(Log log, boolean verbose, Integer bufferSize, String charset, String suffix,
076            boolean nosuffix, boolean skipMerge, boolean skipMinify, String webappSourceDir, String webappTargetDir,
077            String inputDir, List<String> sourceFiles, List<String> sourceIncludes, List<String> sourceExcludes,
078            String outputDir, String outputFilename, Engine engine, YuiConfig yuiConfig, ClosureConfig closureConfig) {
079        super(log, verbose, bufferSize, charset, suffix, nosuffix, skipMerge, skipMinify, webappSourceDir,
080                webappTargetDir, inputDir, sourceFiles, sourceIncludes, sourceExcludes, outputDir, outputFilename,
081                engine, yuiConfig);
082
083        this.closureConfig = closureConfig;
084    }
085
086    /**
087     * Minifies a JavaScript file.
088     *
089     * @param mergedFile input file resulting from the merged step
090     * @param minifiedFile output file resulting from the minify step
091     * @throws IOException when the minify step fails
092     */
093    @Override
094    protected void minify(File mergedFile, File minifiedFile) throws IOException {
095        try (InputStream in = new FileInputStream(mergedFile);
096                OutputStream out = new FileOutputStream(minifiedFile);
097                InputStreamReader reader = new InputStreamReader(in, charset);
098                OutputStreamWriter writer = new OutputStreamWriter(out, charset)) {
099            log.info("Creating the minified file [" + ((verbose) ? minifiedFile.getPath() : minifiedFile.getName())
100                    + "].");
101
102            switch (engine) {
103                case CLOSURE:
104                    log.debug("Using Google Closure Compiler engine.");
105
106                    CompilerOptions options = new CompilerOptions();
107                    closureConfig.getCompilationLevel().setOptionsForCompilationLevel(options);
108                    options.setOutputCharset(charset);
109                    options.setLanguageIn(closureConfig.getLanguage());
110
111                    SourceFile input = SourceFile.fromInputStream(mergedFile.getName(), in);
112                    List<SourceFile> externs = closureConfig.getExterns();
113
114                    Compiler compiler = new Compiler();
115                    compiler.compile(externs, Lists.newArrayList(input), options);
116
117                    if (compiler.hasErrors()) {
118                        throw new EvaluatorException(compiler.getErrors()[0].description);
119                    }
120
121                    writer.append(compiler.toSource());
122                    break;
123                case YUI:
124                    log.debug("Using YUI Compressor engine.");
125
126                    JavaScriptCompressor compressor = new JavaScriptCompressor(reader, new JavaScriptErrorReporter(log,
127                            mergedFile.getName()));
128                    compressor.compress(writer, yuiConfig.getLinebreak(), yuiConfig.isMunge(), verbose,
129                            yuiConfig.isPreserveAllSemiColons(), yuiConfig.isDisableOptimizations());
130                    break;
131                default:
132                    log.warn("JavaScript engine not supported.");
133                    break;
134            }
135        } catch (IOException e) {
136            log.error("Failed to compress the JavaScript file [" + mergedFile.getName() + "].", e);
137            throw e;
138        }
139
140        logCompressionGains(mergedFile, minifiedFile);
141    }
142}
