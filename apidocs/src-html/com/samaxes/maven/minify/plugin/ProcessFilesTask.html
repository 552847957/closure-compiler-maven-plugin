021package com.samaxes.maven.minify.plugin;
022
023import java.io.File;
024import java.io.FileInputStream;
025import java.io.FileOutputStream;
026import java.io.IOException;
027import java.io.InputStream;
028import java.io.InputStreamReader;
029import java.io.OutputStream;
030import java.io.OutputStreamWriter;
031import java.io.SequenceInputStream;
032import java.util.ArrayList;
033import java.util.Collections;
034import java.util.List;
035import java.util.concurrent.Callable;
036import java.util.zip.GZIPOutputStream;
037
038import org.apache.maven.plugin.logging.Log;
039import org.codehaus.plexus.util.DirectoryScanner;
040import org.codehaus.plexus.util.FileUtils;
041import org.codehaus.plexus.util.IOUtil;
042
043import com.samaxes.maven.minify.common.FilenameComparator;
044import com.samaxes.maven.minify.common.ListOfFiles;
045
046/**
047 * Abstract class for merging and compressing a files list.
048 */
049public abstract class ProcessFilesTask implements Callable<Object> {
050
051    public static final String TEMP_SUFFIX = ".tmp";
052
053    protected final Log log;
054
055    protected final Integer bufferSize;
056
057    protected final boolean debug;
058
059    protected final boolean skipMerge;
060
061    protected final boolean skipMinify;
062
063    protected final String charset;
064
065    protected final int linebreak;
066
067    protected final boolean nosuffix;
068
069    protected final String suffix;
070
071    private final String mergedFilename;
072
073    private final File targetDir;
074
075    private final boolean sourceFilesEmpty;
076
077    private final boolean sourceIncludesEmpty;
078
079    private final List<File> files = new ArrayList<File>();
080
081    /**
082     * Task constructor.
083     *
084     * @param log Maven plugin log
085     * @param bufferSize size of the buffer used to read source files
086     * @param debug show source file paths in log output
087     * @param skipMerge whether to skip the merge step or not
088     * @param skipMinify whether to skip the minify step or not
089     * @param webappSourceDir web resources source directory
090     * @param webappTargetDir web resources target directory
091     * @param inputDir directory containing source files
092     * @param sourceFilenames list of source files to include
093     * @param sourceIncludes list of source files to include
094     * @param sourceExcludes list of source files to exclude
095     * @param outputDir directory to write the final file
096     * @param outputFilename the output file name
097     * @param suffix final filename suffix
098     * @param nosuffix whether to use a suffix for the minified filename or not
099     * @param charset if a character set is specified, a byte-to-char variant allows the encoding to be selected.
100     *        Otherwise, only byte-to-byte operations are used
101     * @param linebreak split long lines after a specific column
102     */
103    public ProcessFilesTask(Log log, Integer bufferSize, boolean debug, boolean skipMerge, boolean skipMinify,
104            String webappSourceDir, String webappTargetDir, String inputDir, List<String> sourceFilenames,
105            List<String> sourceIncludes, List<String> sourceExcludes, String outputDir, String outputFilename,
106            String suffix, boolean nosuffix, String charset, int linebreak) {
107        this.log = log;
108        this.bufferSize = bufferSize;
109        this.debug = debug;
110        this.skipMerge = skipMerge;
111        this.skipMinify = skipMinify;
112        this.charset = charset;
113        this.linebreak = linebreak;
114        this.nosuffix = nosuffix;
115        this.mergedFilename = outputFilename;
116        this.suffix = suffix + ".";
117
118        File sourceDir = new File(webappSourceDir + File.separator + inputDir);
119        for (String sourceFilename : sourceFilenames) {
120            addNewSourceFile(mergedFilename, sourceDir, sourceFilename);
121        }
122        for (File sourceInclude : getFilesToInclude(sourceDir, sourceIncludes, sourceExcludes)) {
123            if (!files.contains(sourceInclude)) {
124                addNewSourceFile(mergedFilename, sourceDir, sourceInclude);
125            }
126        }
127
128        this.targetDir = new File(webappTargetDir + File.separator + outputDir);
129        this.sourceFilesEmpty = sourceFilenames.isEmpty();
130        this.sourceIncludesEmpty = sourceIncludes.isEmpty();
131    }
132
133    /**
134     * Method executed by the thread.
135     *
136     * @throws IOException when the merge or minify steps fail
137     */
138    @Override
139    public Object call() throws IOException {
140        synchronized (log) {
141            String fileType = (this instanceof ProcessCSSFilesTask) ? "CSS" : "JavaScript";
142            log.info("Starting " + fileType + " task:");
143
144            if (!files.isEmpty() && (targetDir.exists() || targetDir.mkdirs())) {
145                if (skipMerge) {
146                    log.info("Skipping the merge step...");
147                    for (File mergedFile : files) {
148                        File minifiedFile = new File(targetDir, (nosuffix) ? mergedFile.getName()
149                                : FileUtils.basename(mergedFile.getName()) + suffix
150                                        + FileUtils.getExtension(mergedFile.getName()));
151                        minify(mergedFile, minifiedFile);
152                    }
153                } else if (skipMinify) {
154                    File mergedFile = new File(targetDir, mergedFilename);
155                    merge(mergedFile);
156                    log.info("Skipping the minify step...");
157                } else {
158                    File mergedFile = new File(targetDir, (nosuffix) ? mergedFilename + TEMP_SUFFIX : mergedFilename);
159                    merge(mergedFile);
160                    File minifiedFile = new File(targetDir, (nosuffix) ? mergedFilename
161                            : FileUtils.basename(mergedFilename) + suffix + FileUtils.getExtension(mergedFilename));
162                    minify(mergedFile, minifiedFile);
163                    if (nosuffix) {
164                        mergedFile.deleteOnExit();
165                    }
166                }
167                log.info("");
168            } else if (!sourceFilesEmpty || !sourceIncludesEmpty) {
169                // 'files' list will be empty if source file paths or names added to the project's POM are invalid.
170                log.error("No valid " + fileType + " source files found to process.");
171            }
172        }
173
174        return null;
175    }
176
177    /**
178     * Merges a list of source files.
179     *
180     * @param mergedFile output file resulting from the merged step
181     * @throws IOException when the merge step fails
182     */
183    protected void merge(File mergedFile) throws IOException {
184        try (InputStream sequence = new SequenceInputStream(new ListOfFiles(log, files, debug));
185                OutputStream out = new FileOutputStream(mergedFile);
186                InputStreamReader sequenceReader = new InputStreamReader(sequence, charset);
187                OutputStreamWriter outWriter = new OutputStreamWriter(out, charset)) {
188            log.info("Creating the merged file [" + ((debug) ? mergedFile.getPath() : mergedFile.getName()) + "].");
189
190            IOUtil.copy(sequenceReader, outWriter, bufferSize);
191        } catch (IOException e) {
192            log.error("Failed to concatenate files.", e);
193            throw e;
194        }
195    }
196
197    /**
198     * Minifies a source file.
199     *
200     * @param mergedFile input file resulting from the merged step
201     * @param minifiedFile output file resulting from the minify step
202     * @throws IOException when the minify step fails
203     */
204    abstract void minify(File mergedFile, File minifiedFile) throws IOException;
205
206    /**
207     * Logs compression gains.
208     *
209     * @param mergedFile input file resulting from the merged step
210     * @param minifiedFile output file resulting from the minify step
211     */
212    void logCompressionGains(File mergedFile, File minifiedFile) {
213        try {
214            File temp = File.createTempFile(minifiedFile.getName(), ".gz");
215
216            try (InputStream in = new FileInputStream(minifiedFile);
217                    OutputStream out = new FileOutputStream(temp);
218                    GZIPOutputStream outGZIP = new GZIPOutputStream(out)) {
219                IOUtil.copy(in, outGZIP, bufferSize);
220            }
221
222            log.info("Uncompressed size: " + mergedFile.length() + " bytes.");
223            log.info("Compressed size: " + minifiedFile.length() + " bytes minified (" + temp.length()
224                    + " bytes gzipped).");
225
226            temp.deleteOnExit();
227        } catch (IOException e) {
228            log.debug("Failed to calculate the gzipped file size.", e);
229        }
230    }
231
232    /**
233     * Logs an addition of a new source file.
234     *
235     * @param finalFilename the final file name
236     * @param sourceDir the sources directory
237     * @param sourceFilename the source file name
238     */
239    private void addNewSourceFile(String finalFilename, File sourceDir, String sourceFilename) {
240        File sourceFile = new File(sourceDir, sourceFilename);
241
242        addNewSourceFile(finalFilename, sourceDir, sourceFile);
243    }
244
245    /**
246     * Logs an addition of a new source file.
247     *
248     * @param finalFilename the final file name
249     * @param sourceDir the sources directory
250     * @param sourceFile the source file
251     */
252    private void addNewSourceFile(String finalFilename, File sourceDir, File sourceFile) {
253        if (sourceFile.exists()) {
254            if (finalFilename.equalsIgnoreCase(sourceFile.getName())) {
255                log.warn("The source file [" + ((debug) ? sourceFile.getPath() : sourceFile.getName())
256                        + "] has the same name as the final file.");
257            }
258            log.debug("Adding source file [" + ((debug) ? sourceFile.getPath() : sourceFile.getName()) + "].");
259            files.add(sourceFile);
260        } else {
261            log.warn("The source file [" + ((debug) ? sourceFile.getPath() : sourceFile.getName())
262                    + "] does not exist.");
263        }
264    }
265
266    /**
267     * Returns the files to copy. Default exclusions are used when the excludes list is empty.
268     *
269     * @param baseDir the base directory to start from
270     * @param includes list of source files to include
271     * @param excludes list of source files to exclude
272     * @return the files to copy
273     */
274    private List<File> getFilesToInclude(File baseDir, List<String> includes, List<String> excludes) {
275        List<File> includedFiles = new ArrayList<File>();
276
277        if (includes != null && !includes.isEmpty()) {
278            DirectoryScanner scanner = new DirectoryScanner();
279
280            scanner.setIncludes(includes.toArray(new String[0]));
281            scanner.setExcludes(excludes.toArray(new String[0]));
282            scanner.addDefaultExcludes();
283            scanner.setBasedir(baseDir);
284            scanner.scan();
285
286            for (String includedFilename : scanner.getIncludedFiles()) {
287                includedFiles.add(new File(baseDir, includedFilename));
288            }
289
290            Collections.sort(includedFiles, new FilenameComparator());
291        }
292
293        return includedFiles;
294    }
295}
