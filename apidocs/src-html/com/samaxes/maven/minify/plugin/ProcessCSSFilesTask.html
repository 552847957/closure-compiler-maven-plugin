019package com.samaxes.maven.minify.plugin;
020
021import java.io.File;
022import java.io.FileInputStream;
023import java.io.FileOutputStream;
024import java.io.IOException;
025import java.io.InputStream;
026import java.io.InputStreamReader;
027import java.io.OutputStream;
028import java.io.OutputStreamWriter;
029import java.util.List;
030
031import org.apache.maven.plugin.logging.Log;
032
033import com.yahoo.platform.yui.compressor.CssCompressor;
034
035/**
036 * Task for merging and compressing CSS files.
037 */
038public class ProcessCSSFilesTask extends ProcessFilesTask {
039
040    /**
041     * Task constructor.
042     *
043     * @param log Maven plugin log
044     * @param bufferSize size of the buffer used to read source files
045     * @param debug show source file paths in log output
046     * @param skipMerge whether to skip the merge step or not
047     * @param skipMinify whether to skip the minify step or not
048     * @param cssEngine minify processor engine selected
049     * @param webappSourceDir web resources source directory
050     * @param webappTargetDir web resources target directory
051     * @param inputDir directory containing source files
052     * @param sourceFiles list of source files to include
053     * @param sourceIncludes list of source files to include
054     * @param sourceExcludes list of source files to exclude
055     * @param outputDir directory to write the final file
056     * @param outputFilename the output file name
057     * @param suffix final filename suffix
058     * @param nosuffix whether to use a suffix for the minified filename or not
059     * @param charset if a character set is specified, a byte-to-char variant allows the encoding to be selected.
060     *        Otherwise, only byte-to-byte operations are used
061     * @param linebreak split long lines after a specific column
062     */
063    public ProcessCSSFilesTask(Log log, Integer bufferSize, boolean debug, boolean skipMerge, boolean skipMinify,
064            String cssEngine, String webappSourceDir, String webappTargetDir, String inputDir,
065            List<String> sourceFiles, List<String> sourceIncludes, List<String> sourceExcludes, String outputDir,
066            String outputFilename, String suffix, boolean nosuffix, String charset, int linebreak) {
067        super(log, bufferSize, debug, skipMerge, skipMinify, cssEngine, webappSourceDir, webappTargetDir, inputDir,
068                sourceFiles, sourceIncludes, sourceExcludes, outputDir, outputFilename, suffix, nosuffix, charset,
069                linebreak);
070    }
071
072    /**
073     * Minifies a CSS file.
074     *
075     * @param mergedFile input file resulting from the merged step
076     * @param minifiedFile output file resulting from the minify step
077     * @throws IOException when the minify step fails
078     */
079    @Override
080    protected void minify(File mergedFile, File minifiedFile) throws IOException {
081        try (InputStream in = new FileInputStream(mergedFile);
082                OutputStream out = new FileOutputStream(minifiedFile);
083                InputStreamReader reader = new InputStreamReader(in, charset);
084                OutputStreamWriter writer = new OutputStreamWriter(out, charset)) {
085            log.info("Creating the minified file [" + ((debug) ? minifiedFile.getPath() : minifiedFile.getName())
086                    + "].");
087
088            if ("yui".equals(engine)) {
089                log.debug("Using YUI Compressor engine.");
090
091                CssCompressor compressor = new CssCompressor(reader);
092                compressor.compress(writer, linebreak);
093            } else {
094                log.warn("CSS engine not supported.");
095            }
096        } catch (IOException e) {
097            log.error("Failed to compress the CSS file [" + mergedFile.getName() + "].", e);
098            throw e;
099        }
100
101        logCompressionGains(mergedFile, minifiedFile);
102    }
103}
