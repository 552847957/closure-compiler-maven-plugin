019package com.samaxes.maven.minify.plugin;
020
021import java.io.File;
022import java.io.FileInputStream;
023import java.io.FileNotFoundException;
024import java.io.FileOutputStream;
025import java.io.IOException;
026import java.io.InputStream;
027import java.io.InputStreamReader;
028import java.io.OutputStream;
029import java.io.OutputStreamWriter;
030import java.util.List;
031
032import org.apache.maven.plugin.logging.Log;
033
034import com.samaxes.maven.minify.common.YuiConfig;
035import com.samaxes.maven.minify.plugin.MinifyMojo.Engine;
036import com.yahoo.platform.yui.compressor.CssCompressor;
037
038/**
039 * Task for merging and compressing CSS files.
040 */
041public class ProcessCSSFilesTask extends ProcessFilesTask {
042
043    /**
044     * Task constructor.
045     *
046     * @param log Maven plugin log
047     * @param verbose display additional info
048     * @param bufferSize size of the buffer used to read source files
049     * @param charset if a character set is specified, a byte-to-char variant allows the encoding to be selected.
050     *        Otherwise, only byte-to-byte operations are used
051     * @param suffix final file name suffix
052     * @param nosuffix whether to use a suffix for the minified file name or not
053     * @param skipMerge whether to skip the merge step or not
054     * @param skipMinify whether to skip the minify step or not
055     * @param webappSourceDir web resources source directory
056     * @param webappTargetDir web resources target directory
057     * @param inputDir directory containing source files
058     * @param sourceFiles list of source files to include
059     * @param sourceIncludes list of source files to include
060     * @param sourceExcludes list of source files to exclude
061     * @param outputDir directory to write the final file
062     * @param outputFilename the output file name
063     * @param engine minify processor engine selected
064     * @param yuiConfig YUI Compressor configuration
065     * @throws FileNotFoundException when the given source file does not exist
066     */
067    public ProcessCSSFilesTask(Log log, boolean verbose, Integer bufferSize, String charset, String suffix,
068            boolean nosuffix, boolean skipMerge, boolean skipMinify, String webappSourceDir, String webappTargetDir,
069            String inputDir, List<String> sourceFiles, List<String> sourceIncludes, List<String> sourceExcludes,
070            String outputDir, String outputFilename, Engine engine, YuiConfig yuiConfig) throws FileNotFoundException {
071        super(log, verbose, bufferSize, charset, suffix, nosuffix, skipMerge, skipMinify, webappSourceDir,
072                webappTargetDir, inputDir, sourceFiles, sourceIncludes, sourceExcludes, outputDir, outputFilename,
073                engine, yuiConfig);
074    }
075
076    /**
077     * Minifies a CSS file.
078     *
079     * @param mergedFile input file resulting from the merged step
080     * @param minifiedFile output file resulting from the minify step
081     * @throws IOException when the minify step fails
082     */
083    @Override
084    protected void minify(File mergedFile, File minifiedFile) throws IOException {
085        try (InputStream in = new FileInputStream(mergedFile);
086                OutputStream out = new FileOutputStream(minifiedFile);
087                InputStreamReader reader = new InputStreamReader(in, charset);
088                OutputStreamWriter writer = new OutputStreamWriter(out, charset)) {
089            log.info("Creating the minified file [" + ((verbose) ? minifiedFile.getPath() : minifiedFile.getName())
090                    + "].");
091
092            switch (engine) {
093                case YUI:
094                    log.debug("Using YUI Compressor engine.");
095
096                    CssCompressor compressor = new CssCompressor(reader);
097                    compressor.compress(writer, yuiConfig.getLineBreak());
098                    break;
099                default:
100                    log.warn("CSS engine not supported.");
101                    break;
102            }
103        } catch (IOException e) {
104            log.error("Failed to compress the CSS file [" + ((verbose) ? mergedFile.getPath() : mergedFile.getName())
105                    + "].", e);
106            throw e;
107        }
108
109        logCompressionGains(mergedFile, minifiedFile);
110    }
111}
