353    /**
354     * The degree of compression and optimization to apply to your JavaScript.<br/>
355     * There are three possible compilation levels:
356     * <ul>
357     * <li>{@code WHITESPACE_ONLY}: Just removes whitespace and comments from your JavaScript.</li>
358     * <li>{@code SIMPLE_OPTIMIZATIONS}: Performs compression and optimization that does not interfere with the
359     * interaction between the compiled JavaScript and other JavaScript. This level renames only local variables.</li>
360     * <li>{@code ADVANCED_OPTIMIZATIONS}: Achieves the highest level of compression by renaming symbols in your
361     * JavaScript. When using {@code ADVANCED_OPTIMIZATIONS} compilation you must perform extra steps to preserve
362     * references to external symbols. See <a href="/closure/compiler/docs/api-tutorial3">Advanced Compilation and
363     * Externs</a> for more information about {@code ADVANCED_OPTIMIZATIONS}.</li>
364     * </ul>
365     *
366     * @since 1.7.2
367     */
368    @Parameter(property = "closureCompilationLevel", defaultValue = "SIMPLE_OPTIMIZATIONS")
369    private CompilationLevel closureCompilationLevel;

374     * {@code closureExterns} parameter only has an effect if you are using a {@code CompilationLevel} of
375     * {@code ADVANCED_OPTIMIZATIONS}.<br/>
376     * These file names are relative to {@link #webappSourceDir} directory.

378     * @since 1.7.2
379     */
380    @Parameter(property = "closureExterns")
381    private ArrayList<String> closureExterns;

383    /**
384     * Executed when the goal is invoked, it will first invoke a parallel lifecycle, ending at the given phase.
385     */
386    @Override
387    public void execute() throws MojoExecutionException, MojoFailureException {
388        checkDeprecatedOptions();
389
390        if (skipMerge && skipMinify) {
391            getLog().warn("Both merge and minify steps are configured to be skipped.");
392            return;
393        }
394
395        fillOptionalValues();
396        YuiConfig yuiConfig = fillYuiConfig();
397        ClosureConfig closureConfig = fillClosureConfig();
398
399        Collection<ProcessFilesTask> processFilesTasks = new ArrayList<ProcessFilesTask>();
400        processFilesTasks.add(new ProcessCSSFilesTask(getLog(), debug, bufferSize, charset, suffix, nosuffix,
401                skipMerge, skipMinify, webappSourceDir, webappTargetDir, cssSourceDir, cssSourceFiles,
402                cssSourceIncludes, cssSourceExcludes, cssTargetDir, cssFinalFile, cssEngine, yuiConfig));
403        processFilesTasks.add(new ProcessJSFilesTask(getLog(), debug, bufferSize, charset, suffix, nosuffix, skipMerge,
404                skipMinify, webappSourceDir, webappTargetDir, jsSourceDir, jsSourceFiles, jsSourceIncludes,
405                jsSourceExcludes, jsTargetDir, jsFinalFile, jsEngine, yuiConfig, closureConfig));
406
407        ExecutorService executor = Executors.newFixedThreadPool(2);
408        try {
409            List<Future<Object>> futures = executor.invokeAll(processFilesTasks);
410            for (Future<Object> future : futures) {
411                try {
412                    future.get();
413                } catch (ExecutionException e) {
414                    throw new MojoFailureException(e.getMessage(), e);
415                }
416            }
417            executor.shutdown();
418        } catch (InterruptedException e) {
419            executor.shutdownNow();
420            throw new MojoFailureException(e.getMessage(), e);
421        }
422    }
423
424    private void checkDeprecatedOptions() {
425        if (debug == null) {
426            debug = verbose;
427        } else {
428            getLog().warn(
429                    "The option 'debug' is deprecated and will be removed on the next version. Use 'verbose' instead.");
430        }
431        if (linebreak == null) {
432            linebreak = yuiLinebreak;
433        } else {
434            getLog().warn(
435                    "The option 'linebreak' is deprecated and will be removed on the next version. Use 'yuiLinebreak' instead.");
436        }
437        if (munge == null) {
438            munge = yuiMunge;
439        } else {
440            getLog().warn(
441                    "The option 'munge' is deprecated and will be removed on the next version. Use 'yuiMunge' instead.");
442        }
443        if (preserveAllSemiColons == null) {
444            preserveAllSemiColons = yuiPreserveAllSemiColons;
445        } else {
446            getLog().warn(
447                    "The option 'preserveAllSemiColons' is deprecated and will be removed on the next version. Use 'yuiPreserveAllSemiColons' instead.");
448        }
449        if (disableOptimizations == null) {
450            disableOptimizations = yuiDisableOptimizations;
451        } else {
452            getLog().warn(
453                    "The option 'disableOptimizations' is deprecated and will be removed on the next version. Use 'yuiDisableOptimizations' instead.");
454        }
455    }
456
457    private void fillOptionalValues() {
458        if (Strings.isNullOrEmpty(cssTargetDir)) {
459            cssTargetDir = cssSourceDir;
460        }
461        if (Strings.isNullOrEmpty(jsTargetDir)) {
462            jsTargetDir = jsSourceDir;
463        }
464    }
465
466    private YuiConfig fillYuiConfig() {
467        return new YuiConfig(linebreak, munge, preserveAllSemiColons, disableOptimizations);
468    }
469
470    private ClosureConfig fillClosureConfig() {
471        List<SourceFile> externs = new ArrayList<>();
472        for (String extern : closureExterns) {
473            externs.add(SourceFile.fromFile(webappSourceDir + File.separator + extern, Charset.forName(charset)));
474        }
475        return new ClosureConfig(closureLanguage, closureCompilationLevel, externs);
476    }
