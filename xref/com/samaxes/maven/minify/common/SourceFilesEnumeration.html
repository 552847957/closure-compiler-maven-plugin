19  package com.samaxes.maven.minify.common;
20  
21  import java.io.File;
22  import java.io.FileInputStream;
23  import java.io.FileNotFoundException;
24  import java.io.InputStream;
25  import java.util.Enumeration;
26  import java.util.List;
27  import java.util.NoSuchElementException;
28  
29  import org.apache.maven.plugin.logging.Log;
30  
31  /**
32   * Used to initialize a {@code SequenceInputStream} with a {@code Enumeration<? extends InputStream>}. The input streams
33   * that are produced by the enumeration will be read, in order, to provide the bytes to be read from the
34   * {@code SequenceInputStream}.
35   */
36  public class SourceFilesEnumeration implements Enumeration<InputStream> {
37  
38      private List<File> files;
39  
40      private int current = 0;
41  
42      /**
43       * Enumeration public constructor.
44       *
45       * @param log Maven plugin log
46       * @param files list of files
47       * @param debug show source file paths in log output
48       */
49      public SourceFilesEnumeration(Log log, List<File> files, boolean debug) {
50          this.files = files;
51  
52          for (File file : files) {
53              log.info("Processing source file [" + ((debug) ? file.getPath() : file.getName()) + "].");
54          }
55      }
56  
57      /**
58       * Tests if this enumeration contains more elements.
59       *
60       * @return {@code true} if and only if this enumeration object contains at least one more element to provide;
61       *         {@code false} otherwise.
62       */
63      @Override
64      public boolean hasMoreElements() {
65          return (current < files.size()) ? true : false;
66      }
67  
68      /**
69       * Returns the next element of this enumeration if this enumeration object has at least one more element to provide.
70       *
71       * @return the next element of this enumeration.
72       * @exception NoSuchElementException if no more elements exist.
73       */
74      @Override
75      public InputStream nextElement() {
76          InputStream is = null;
77  
78          if (!hasMoreElements()) {
79              throw new NoSuchElementException("No more files!");
80          } else {
81              File nextElement = files.get(current);
82              current++;
83  
84              try {
85                  is = new FileInputStream(nextElement);
86              } catch (FileNotFoundException e) {
87                  throw new NoSuchElementException("The path [" + nextElement.getPath() + "] cannot be found.");
88              }
89          }
90  
91          return is;
92      }
93  }
