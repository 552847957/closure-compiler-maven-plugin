357     /**
358      * The degree of compression and optimization to apply to your JavaScript.<br/>
359      * There are three possible compilation levels:
360      * <ul>
361      * <li>{@code WHITESPACE_ONLY}: Just removes whitespace and comments from your JavaScript.</li>
362      * <li>{@code SIMPLE_OPTIMIZATIONS}: Performs compression and optimization that does not interfere with the
363      * interaction between the compiled JavaScript and other JavaScript. This level renames only local variables.</li>
364      * <li>{@code ADVANCED_OPTIMIZATIONS}: Achieves the highest level of compression by renaming symbols in your
365      * JavaScript. When using {@code ADVANCED_OPTIMIZATIONS} compilation you must perform extra steps to preserve
366      * references to external symbols. See <a href="/closure/compiler/docs/api-tutorial3">Advanced Compilation and

407     /**
408      * Executed when the goal is invoked, it will first invoke a parallel lifecycle, ending at the given phase.
409      */
410     @Override
411     public void execute() throws MojoExecutionException, MojoFailureException {
412         checkDeprecatedOptions();
413 
414         if (skipMerge && skipMinify) {
415             getLog().warn("Both merge and minify steps are configured to be skipped.");
416             return;
417         }
418 
419         fillOptionalValues();
420         YuiConfig yuiConfig = fillYuiConfig();
421         ClosureConfig closureConfig = fillClosureConfig();
422 
423         Collection<ProcessFilesTask> processFilesTasks = new ArrayList<ProcessFilesTask>();
424         try {
425             processFilesTasks.add(new ProcessCSSFilesTask(getLog(), debug, bufferSize, charset, suffix, nosuffix,
426                     skipMerge, skipMinify, webappSourceDir, webappTargetDir, cssSourceDir, cssSourceFiles,
427                     cssSourceIncludes, cssSourceExcludes, cssTargetDir, cssFinalFile, cssEngine, yuiConfig));
428             processFilesTasks.add(new ProcessJSFilesTask(getLog(), debug, bufferSize, charset, suffix, nosuffix,
429                     skipMerge, skipMinify, webappSourceDir, webappTargetDir, jsSourceDir, jsSourceFiles,
430                     jsSourceIncludes, jsSourceExcludes, jsTargetDir, jsFinalFile, jsEngine, yuiConfig, closureConfig));
431         } catch (FileNotFoundException e) {
432             throw new MojoFailureException(e.getMessage(), e);
433         }
434 
435         ExecutorService executor = Executors.newFixedThreadPool(2);
436         try {
437             List<Future<Object>> futures = executor.invokeAll(processFilesTasks);
438             for (Future<Object> future : futures) {
439                 try {
440                     future.get();
441                 } catch (ExecutionException e) {
442                     throw new MojoExecutionException(e.getMessage(), e);
443                 }
444             }
445             executor.shutdown();
446         } catch (InterruptedException e) {
447             executor.shutdownNow();
448             throw new MojoExecutionException(e.getMessage(), e);
449         }
450     }

452     private void checkDeprecatedOptions() {
453         if (debug == null) {
454             debug = verbose;
455         } else {
456             getLog().warn(
457                     "The option 'debug' is deprecated and will be removed on the next version. Use 'verbose' instead.");
458         }
459         if (linebreak == null) {
460             linebreak = yuiLineBreak;
461         } else {
462             getLog().warn(
463                     "The option 'linebreak' is deprecated and will be removed on the next version. Use 'yuiLineBreak' instead.");
464         }
465         if (munge == null) {
466             munge = !yuiNoMunge;
467         } else {
468             getLog().warn(
469                     "The option 'munge' is deprecated and will be removed on the next version. Use 'yuiNoMunge' instead.");
470         }
471         if (preserveAllSemiColons == null) {
472             preserveAllSemiColons = yuiPreserveSemicolons;
473         } else {
474             getLog().warn(
475                     "The option 'preserveAllSemiColons' is deprecated and will be removed on the next version. Use 'yuiPreserveSemicolons' instead.");
476         }
477         if (disableOptimizations == null) {
478             disableOptimizations = yuiDisableOptimizations;
479         } else {
480             getLog().warn(
481                     "The option 'disableOptimizations' is deprecated and will be removed on the next version. Use 'yuiDisableOptimizations' instead.");
482         }
483     }
484 
485     private void fillOptionalValues() {
486         if (Strings.isNullOrEmpty(cssTargetDir)) {
487             cssTargetDir = cssSourceDir;
488         }
489         if (Strings.isNullOrEmpty(jsTargetDir)) {
490             jsTargetDir = jsSourceDir;
491         }
492         if (Strings.isNullOrEmpty(charset)) {
493             charset = Charset.defaultCharset().name();
494         }
495     }
496 
497     private YuiConfig fillYuiConfig() {
498         return new YuiConfig(linebreak, munge, preserveAllSemiColons, disableOptimizations);
499     }
500 
501     private ClosureConfig fillClosureConfig() {
502         List<SourceFile> externs = new ArrayList<>();
503         for (String extern : closureExterns) {
504             externs.add(SourceFile.fromFile(webappSourceDir + File.separator + extern, Charset.forName(charset)));
505         }
506         return new ClosureConfig(closureLanguage, closureCompilationLevel, externs, closureCreateSourceMap,
507                 closureAngularPass);
508     }
